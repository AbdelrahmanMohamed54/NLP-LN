import re
import nltk

# Read the corpus file
file_path = 'deu_news_2023_100K-sentences.txt'
with open(file_path, 'r', encoding='utf-8') as file:
    corpus = file.read()

# Remove line numbers and special characters
lines = corpus.split('\n')
processed_lines = [re.sub(r'^\d+\t', '', line) for line in lines if line]
processed_text = ' '.join(processed_lines)

# Remove URLs
processed_text = re.sub(r'http\S+|www\S+', '', processed_text)

# Remove non-alphanumeric characters and digits, except for letters
processed_text = re.sub(r'[^A-Za-zäöüÄÖÜß\s]', '', processed_text)

# Remove extra spaces
processed_text = re.sub(r'\s+', ' ', processed_text).strip()

# Tokenize words
words = nltk.word_tokenize(processed_text)

# Filter out non-alphabetic tokens
words = [word for word in words if word.isalpha()]

# Rejoin the words into a single string
processed_text = ' '.join(words)

print(processed_text[:1000])  # Print first 1000 characters of the cleaned text
